set number
set expandtab
set ts=4 sw=4
set softtabstop=4

" Status Line
set showcmd
set laststatus=2

" Search
set ignorecase
set smartcase
set incsearch
set hlsearch

set showmatch
set matchtime=1  " set to 100[msec], default is 5
set backspace=2

" Assign jj as ESC in insert mode.
inoremap <silent> jj <ESC>

" yank whole line
nnoremap Y y$

" show tag candidate
nnoremap <C-]> g<C-]>

colorscheme koehler
